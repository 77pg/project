        // 在页面加载完成后执行操作
        window.onload = function () {
            // 获取英文内容的元素
            var englishContent = document.getElementById("englishContent");

            // 隐藏英文内容
            englishContent.style.display = "none";

            // 创建"More"按钮元素
            var moreButton = document.createElement("button");
            moreButton.innerHTML = "More";
            moreButton.classList.add("btn", "btn-primary", "mt-3");

            // 插入"More"按钮到英文内容后面
            englishContent.insertAdjacentElement("afterend", moreButton);

            // "More"按钮点击事件处理函数
            moreButton.onclick = function () {
                if (englishContent.style.display === "none") {
                    // 显示英文内容
                    englishContent.style.display = "block";
                    moreButton.innerHTML = "收起";
                } else {
                    // 隐藏英文内容
                    englishContent.style.display = "none";
                    moreButton.innerHTML = "More";
                }
            };
        };
    </script>
